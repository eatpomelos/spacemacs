** 优化packge系统
- 配置packge系统来优化emacs
#+BEGIN_SRC emacs-lisp
(when (>= emacs-major-version 24)
  (require 'package)
  (package-initialize)
  (add-to-list 'package-archives '("gun" . "http://elpa.emacs-china.org/gnu/") t)
  (add-to-list 'package-archives '("melpa" . "http://melpa.org/packages/") t)
  )

(require 'cl)

;;add whatever package you want here
(defvar pomeloly/packages '(
			    company
			    zenburn-theme
			    monokai-theme
			    hungry-delete
			    ) "Default packages")

(defun pomeloly/packages-installed-p ()
  (loop for pkg in pomeloly/packages
	when (not (package-installed-p pkg)) do (return nil)
	finally (return t)))

(unless (pomeloly/packages-installed-p)
  (message "%s" "Refreshing package database...")
  (package-refresh-contents)
  (dolist (pkg pomeloly/packages)
    (when (not (package-installed-p pkg))
      (package-install pkg))))

#+END_SRC
